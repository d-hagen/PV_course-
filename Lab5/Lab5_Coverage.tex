\section*{FIFO}

\subsection*{Task 1}

\subsubsection*{Question 1.1: Covergroup Bins}

The testbench defines one \texttt{covergroup} (\texttt{fifo\_cov}) and one \texttt{cover property} directive. In total there are \textbf{13 covergroup bins + 1 cover directive}.

\paragraph*{Coverpoint \texttt{full} (1 bin)}
\begin{enumerate}[label=\arabic*.]
  \item \texttt{went\_full} --- bin for \texttt{full == 1}
\end{enumerate}

\paragraph*{Coverpoint \texttt{empty} (1 bin)}
\begin{enumerate}[resume, label=\arabic*.]
  \item \texttt{went\_empty} --- bin for \texttt{empty == 1}
\end{enumerate}

\paragraph*{Coverpoint \texttt{dut.count} (3 bins)}
\begin{enumerate}[resume, label=\arabic*.]
  \item \texttt{empty\_bin} --- occupancy equals 0
  \item \texttt{mid\_bins} --- occupancy in range \texttt{[1 : DEPTH-1]}
  \item \texttt{full\_bin} --- occupancy equals \texttt{DEPTH}
\end{enumerate}

\paragraph*{Coverpoint \texttt{wr\_en} (2 auto-bins)}
\begin{enumerate}[resume, label=\arabic*.]
  \item \texttt{auto[0]} --- \texttt{wr\_en == 0}
  \item \texttt{auto[1]} --- \texttt{wr\_en == 1}
\end{enumerate}

\paragraph*{Coverpoint \texttt{rd\_en} (2 auto-bins)}
\begin{enumerate}[resume, label=\arabic*.]
  \item \texttt{auto[0]} --- \texttt{rd\_en == 0}
  \item \texttt{auto[1]} --- \texttt{rd\_en == 1}
\end{enumerate}

\paragraph*{Cross \texttt{wr\_en} $\times$ \texttt{rd\_en} (4 bins)}
\begin{enumerate}[resume, label=\arabic*.]
  \item \texttt{write\_only} --- \texttt{wr\_en=1, rd\_en=0}
  \item \texttt{read\_only} --- \texttt{wr\_en=0, rd\_en=1}
  \item \texttt{both\_off} --- \texttt{wr\_en=0, rd\_en=0}
  \item \texttt{illegal} --- \texttt{wr\_en=1, rd\_en=1}
\end{enumerate}

\paragraph*{Cover Directive (+1)}
\begin{enumerate}[resume, label={D1.}]
  \item \texttt{cover property (becomes\_empty\_after\_read)} --- covers the sequence where the FIFO transitions to empty after a read when \texttt{dut.count > 0}.
\end{enumerate}

\subsubsection*{Question 1.2: Bins Covered After Reset Only}

The following \textbf{5 bins} are covered just from reset:

\begin{enumerate}
  \item \texttt{went\_empty} --- after reset the FIFO is empty, so \texttt{empty == 1}
  \item \texttt{empty\_bin} --- \texttt{dut.count == 0} after reset
  \item \texttt{wr\_en auto[0]} --- \texttt{wr\_en} stays 0
  \item \texttt{rd\_en auto[0]} --- \texttt{rd\_en} stays 0
  \item \texttt{both\_off} --- cross bin where \texttt{wr\_en=0, rd\_en=0}
\end{enumerate}

\subsection*{Task 2}

\subsubsection*{Question 2.1: Additionally Covered Bins}

After enabling the provided stimulus, the following \textbf{5 additional bins} are now covered:

\begin{enumerate}
  \item \texttt{write\_only} --- cross bin where \texttt{wr\_en=1, rd\_en=0}
  \item \texttt{read\_only} --- cross bin where \texttt{wr\_en=0, rd\_en=1}
  \item \texttt{wr\_en auto[1]} --- \texttt{wr\_en == 1}
  \item \texttt{rd\_en auto[1]} --- \texttt{rd\_en == 1}
  \item \texttt{mid\_bins} --- occupancy in range \texttt{[1 : DEPTH-1]}
\end{enumerate}

\subsection*{Task 3}

\subsubsection*{Missing Bins After Task 2}

The following \textbf{3 bins + 1 cover directive} remained uncovered:

\begin{enumerate}
  \item \texttt{went\_full} --- \texttt{full == 1}
  \item \texttt{full\_bin} --- \texttt{dut.count == DEPTH}
  \item \texttt{illegal} --- cross bin where \texttt{wr\_en=1, rd\_en=1}
\end{enumerate}
\begin{enumerate}[label={D1.}]
  \item \texttt{becomes\_empty\_after\_read} --- FIFO transitions to empty after a read when \texttt{dut.count > 0}
\end{enumerate}

\subsubsection*{Additional Stimulus}

\paragraph*{Fully fill and fully empty the FIFO.}
Increased the number of consecutive writes to exceed \texttt{DEPTH} (16), driving the FIFO to full occupancy, then increased the number of consecutive reads to drain it completely back to empty.

\textbf{Hits bins:}
\begin{itemize}
  \item \texttt{went\_full} --- FIFO reaches full capacity
  \item \texttt{full\_bin} --- \texttt{dut.count == DEPTH}
  \item \texttt{becomes\_empty\_after\_read} (D1) --- FIFO drains to empty after a read while \texttt{dut.count > 0}
\end{itemize}

\paragraph*{Drive both \texttt{wr\_en} and \texttt{rd\_en} on the same cycle.}
Added stimulus that asserts \texttt{wr\_en=1} and \texttt{rd\_en=1} simultaneously.

\textbf{Hits bin:}
\begin{itemize}
  \item \texttt{illegal} --- cross bin where \texttt{wr\_en=1, rd\_en=1}
\end{itemize}

\section*{Cache}

\subsection*{Task 4}

\subsubsection*{Covergroup Bins}

The testbench defines one \texttt{covergroup} (\texttt{cache\_cov}) and six \texttt{cover property} directives. In total there are \textbf{25 covergroup bins + 6 cover directives}.

\paragraph*{Coverpoint \texttt{write} (2 auto-bins)}
\begin{enumerate}[label=\arabic*.]
  \item \texttt{auto[0]} --- \texttt{write == 0}
  \item \texttt{auto[1]} --- \texttt{write == 1}
\end{enumerate}

\paragraph*{Coverpoint \texttt{read} (2 auto-bins)}
\begin{enumerate}[resume, label=\arabic*.]
  \item \texttt{auto[0]} --- \texttt{read == 0}
  \item \texttt{auto[1]} --- \texttt{read == 1}
\end{enumerate}

\paragraph*{Coverpoint \texttt{hit} (4 transition bins)}
\begin{enumerate}[resume, label=\arabic*.]
  \item \texttt{zero\_to\_one} --- hit transitions $0 \Rightarrow 1$ (miss to hit)
  \item \texttt{one\_to\_zero} --- hit transitions $1 \Rightarrow 0$ (hit to miss)
  \item \texttt{stays\_hit} --- hit stays $1 \Rightarrow 1$ (consecutive hits)
  \item \texttt{stays\_miss} --- hit stays $0 \Rightarrow 0$ (consecutive misses)
\end{enumerate}

\paragraph*{Cross \texttt{write} $\times$ \texttt{read} (4 bins)}
\begin{enumerate}[resume, label=\arabic*.]
  \item \texttt{write\_only} --- \texttt{write=1, read=0}
  \item \texttt{read\_only} --- \texttt{write=0, read=1}
  \item \texttt{both\_off} --- \texttt{write=0, read=0}
  \item \texttt{read\_over\_write} --- \texttt{write=1, read=1} (read path takes priority)
\end{enumerate}

\paragraph*{Cross \texttt{write} $\times$ \texttt{hit} (2 bins)}
\begin{enumerate}[resume, label=\arabic*.]
  \item \texttt{write\_hit} --- write with cache hit
  \item \texttt{write\_miss} --- write with cache miss
\end{enumerate}

\paragraph*{Cross \texttt{read} $\times$ \texttt{hit} (2 bins)}
\begin{enumerate}[resume, label=\arabic*.]
  \item \texttt{read\_hit} --- read with cache hit
  \item \texttt{read\_miss} --- read with cache miss
\end{enumerate}

\paragraph*{Coverpoint \texttt{dut.tag} (3 bins)}
\begin{enumerate}[resume, label=\arabic*.]
  \item \texttt{lower\_bins} --- tag in range \texttt{[0 : TAG\_LO]}
  \item \texttt{mid\_bins} --- tag in range \texttt{[TAG\_LO+1 : TAG\_HI-1]}
  \item \texttt{high\_bins} --- tag in range \texttt{[TAG\_HI : TAG\_MAX]}
\end{enumerate}

\paragraph*{Coverpoint \texttt{dut.index} (3 bins)}
\begin{enumerate}[resume, label=\arabic*.]
  \item \texttt{lower\_bins} --- index in range \texttt{[0 : INDEX\_LO]}
  \item \texttt{mid\_bins} --- index in range \texttt{[INDEX\_LO+1 : INDEX\_HI-1]}
  \item \texttt{high\_bins} --- index in range \texttt{[INDEX\_HI : INDEX\_MAX]}
\end{enumerate}

\paragraph*{Coverpoint \texttt{dut.offset} (3 bins)}
\begin{enumerate}[resume, label=\arabic*.]
  \item \texttt{lower\_bins} --- offset in range \texttt{[0 : OFFSET\_LO]}
  \item \texttt{mid\_bins} --- offset in range \texttt{[OFFSET\_LO+1 : OFFSET\_HI-1]}
  \item \texttt{high\_bins} --- offset in range \texttt{[OFFSET\_HI : OFFSET\_MAX]}
\end{enumerate}

\paragraph*{Cover Directives (+6)}
\begin{enumerate}[label={D\arabic*.}]
  \item \texttt{p\_read\_replace} --- read to a valid line with tag mismatch (conflict miss / replacement)
  \item \texttt{p\_write\_replace} --- write to a valid line with tag mismatch (conflict miss / replacement)
  \item \texttt{p\_read\_cold\_miss} --- read to an invalid line (compulsory miss)
  \item \texttt{p\_write\_cold\_miss} --- write to an invalid line (compulsory miss)
  \item \texttt{p\_write\_then\_read\_hit} --- write followed by read hit to the same address (data round-trip)
  \item \texttt{p\_read\_miss\_then\_hit} --- read miss followed by read hit to the same address (cache fill path)
\end{enumerate}
